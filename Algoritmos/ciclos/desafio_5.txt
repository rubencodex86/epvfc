Algoritmo tres_tres

Inicio

var contador :inteiro;

    contador <- 0;

    Enquanto (contador <= 6 )
        Escrever (3*contador);
        contador <- contador+1;
    FimEnquanto

Algoritmo dez_tres

Inicio

var contador :inteiro;

    contador <- 10;

    Enquanto (contador >= 3)
        Escrever (contador);
        contador <- contador-1;
    FimEnquanto
Fim
